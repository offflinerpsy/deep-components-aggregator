# Добавить в ~/.ssh/config на локальной машине:

Host deep-prod
  HostName 89.104.69.77
  User root
  Port 22
  ServerAliveInterval 30
  ServerAliveCountMax 4
  TCPKeepAlive yes
  ControlMaster auto
  ControlPersist 10m
  ControlPath ~/.ssh/cm-%r@%h:%p

# Для использования:
# ssh deep-prod
# rsync -e "ssh deep-prod" ...
